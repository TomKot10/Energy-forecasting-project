# ============= METRYKI =============
def eval_metrics(actual, pred):
    actual = np.asarray(actual); pred = np.asarray(pred)
    me = np.mean(pred - actual)
    mae = np.mean(np.abs(pred - actual))
    with np.errstate(divide='ignore', invalid='ignore'):
        mape = 100 * np.nanmean(np.where(np.abs(actual) > 1e-8, np.abs((pred - actual) / actual), np.nan))
    rmse = np.sqrt(np.mean((pred - actual) ** 2))
    rmspe = 100 * (rmse / np.mean(np.abs(actual))) if np.mean(np.abs(actual)) > 1e-8 else np.nan
    theil = np.nan
    try:
        theil = np.sqrt(np.mean(((pred[1:] - actual[1:]) ** 2)) / np.mean(actual[1:] ** 2))
    except Exception:
        pass
    return {"ME": me, "MAE": mae, "MAPE": mape, "RMSE": rmse, "RMSPE": rmspe, "Theil": theil}

# ============= NARZĘDZIA MODELOWE =============
def fit_hw(y_train, h):
    m = ExponentialSmoothing(y_train, trend="add", seasonal=None).fit()
    fc = m.forecast(h)
    return {"name": "HW", "model": m, "fc": fc, "fitted": m.fittedvalues, "orders": None}

def fit_arima(y_train, h, order=(1,1,1)):
    m = SARIMAX(y_train, order=order, enforce_stationarity=False, enforce_invertibility=False).fit(disp=False)
    res = m.get_forecast(steps=h)
    fc = res.predicted_mean
    conf = res.conf_int(alpha=0.05)
    return {"name": f"ARIMA{order}", "model": m, "fc": fc, "fitted": m.fittedvalues, "conf": conf, "orders": ("ARIMA", order, None)}

def fit_sarima(y_train, h, order=(1,1,1), seasonal_order=(1,1,1,12)):
    m = SARIMAX(y_train, order=order, seasonal_order=seasonal_order,
                enforce_stationarity=False, enforce_invertibility=False).fit(disp=False)
    res = m.get_forecast(steps=h)
    fc = res.predicted_mean
    conf = res.conf_int(alpha=0.05)
    return {"name": f"SARIMA{order}x{seasonal_order}", "model": m, "fc": fc, "fitted": m.fittedvalues, "conf": conf, "orders": ("SARIMA", order, seasonal_order)}

def fit_autoarima(y_train, h):
    from pmdarima import auto_arima
    aa = auto_arima(
        y_train, start_p=0, start_q=0, max_p=3, max_q=3,
        start_P=0, start_Q=0, max_P=2, max_Q=2, seasonal=True, m=12,
        max_d=2, max_D=1, information_criterion="aicc", stepwise=True,
        with_intercept=True, trace=False, error_action="ignore", suppress_warnings=True
    )
    fc = pd.Series(aa.predict(n_periods=h), index=pd.date_range(y_train.index[-1] + pd.offsets.MonthBegin(1), periods=h, freq="MS"))
    # do PI dopasujemy SARIMAX na tych samych rzędach
    try:
        m = SARIMAX(y_train, order=aa.order, seasonal_order=aa.seasonal_order,
                    enforce_stationarity=False, enforce_invertibility=False).fit(disp=False)
        res = m.get_forecast(steps=h)
        conf = res.conf_int(alpha=0.05)
        fitted = m.fittedvalues
        return {"name": f"AUTOARIMA{aa.order}x{aa.seasonal_order}", "model": m, "fc": fc, "fitted": fitted, "conf": conf, "orders": ("SARIMA", aa.order, aa.seasonal_order)}
    except Exception:
        return {"name": f"AUTOARIMA{aa.order}x{aa.seasonal_order}", "model": aa, "fc": fc, "fitted": None, "orders": ("AUTOARIMA", aa.order, aa.seasonal_order)}

# ============= PODEJŚCIE 2: PROGNOZA PRZEZ SKŁADNIKI (HW/ARIMA/SARIMA/AUTOARIMA) =============

        fc = pd.Series([last]*h, index=df_test.index)
        pred_components[col] = fc
        component_choice[col] = "Fallback-last"
        component_orders[col] = None
        component_metrics[col] = {"RMSE": np.nan, "MAPE": np.nan, "RMSPE": np.nan}
        continue

    # Kandydaci
    C = []
    C.append(fit_hw(ytr, h))
    C.append(fit_arima(ytr, h, order=(1,1,1)))
    C.append(fit_sarima(ytr, h, order=(1,1,1), seasonal_order=(1,1,1,12)))
    if USE_AUTOARIMA:
        try:
            C.append(fit_autoarima(ytr, h))
        except Exception:
            pass

    # Ocena na zbiorze testowym (h-krokowa prognoza out-of-sample)
    best_name, best_fc, best_orders, best_rmse = None, None, None, np.inf
    for c in C:
        fc = c["fc"].reindex(df_test.index)
        m = eval_metrics(df_test[col].values, fc.values)
        rmse = m["RMSE"]
        # zapisz metryki kandydata (opcjonalnie)
        # wybierz najlepszy po RMSE
        if rmse < best_rmse:
            best_rmse = rmse
            best_name = c["name"]
            best_fc = fc
            best_orders = c.get("orders", None)

    pred_components[col] = best_fc
    component_choice[col] = best_name
    component_orders[col] = best_orders
    # policz końcowe metryki dla zwycięzcy
    component_metrics[col] = {k: round(v, 3) if v is not None else np.nan for k, v in eval_metrics(df_test[col].values, best_fc.values).items()}

# 1) EX-ANTE PRZEZ SKŁADNIKI — refit każdy składnik najlepszym typem modelu
def refit_and_forecast_component(series, orders_info, h):
    """Refit komponentu na CAŁYM szeregu danym typem wybranym w walidacji."""
    name = component_choice[series.name]
    s = series.dropna()
    if orders_info is None or name == "HW":
        m = ExponentialSmoothing(s, trend="add", seasonal=None).fit()
        return pd.Series(m.forecast(h).values, index=fc_index), ("HW", None, None)
    kind, order, seasonal_order = orders_info if orders_info is not None else ("HW", None, None)
    if kind == "ARIMA":
        m = SARIMAX(s, order=order, enforce_stationarity=False, enforce_invertibility=False).fit(disp=False)
        return pd.Series(m.forecast(h).values, index=fc_index), ("ARIMA", order, None)
    elif kind == "SARIMA" or kind == "AUTOARIMA":
        # AUTOARIMA została zmapowana na SARIMAX dla PI, więc traktujemy jak SARIMA
        seas = seasonal_order if seasonal_order is not None else (0,0,0,0)
        m = SARIMAX(s, order=order, seasonal_order=seas, enforce_stationarity=False, enforce_invertibility=False).fit(disp=False)
        return pd.Series(m.forecast(h).values, index=fc_index), ("SARIMA", order, seas)
    else:
        # bezpieczny fallback
        last = s.iloc[-1] if len(s) else 0.0
        return pd.Series([last]*h, index=fc_index), ("Fallback", None, None)

ex_ante_components = []
for col in df.columns:
    fc_col, used = refit_and_forecast_component(df[col], component_orders.get(col), H_FINAL)
    fc_col.name = col
    ex_ante_components.append(fc_col)

def _plot_final(series, fc, title):
    fig, ax = plt.subplots(figsize=(11,4))
    ax.plot(series.index, series.values, lw=1, label="obserwacje (pełny okres)")
    ax.plot(fc.index, fc.values, "r--o", lw=1, label="prognoza ex-ante")
    ax.axvspan(fc.index[0], fc.index[-1], color="gray", alpha=0.12)
    ax.set_xlim(series.index.min(), fc.index[-1])
    ax.xaxis.set_major_locator(mdates.YearLocator(1))
    ax.xaxis.set_major_formatter(mdates.DateFormatter("%Y-%m"))
    if INVERT_Y: ax.invert_yaxis()
    plt.xticks(rotation=45)
    ax.set_title(title)
    ax.legend(); ax.grid(alpha=0.3); plt.tight_layout(); plt.show()

_plot_final(idx_ew,  fc_ex_ew,  "POTWREG (EW) — ex-ante przez składniki")
